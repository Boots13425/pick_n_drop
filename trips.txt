 const Form = document.querySelector('form')
        
        Form.addEventListener('submit', async(e)=>{
            const data = {
                username: Form.elements['username'].value,
                Date: Form.elements['Date'].value,
                Time: Form.elements['Time'].value,
                Location: Form.elements['Location'].value,
                Destination: Form.elements['Destination'].value,
                Items: Form.elements['Items'].value
            }
            
            const response = await fetch('/postTrip', {
                method: "POST",
                headers: {
                    "Content-Type":"application/json"
                },
                body:JSON.stringify(data)
            })

            .then(res => res.json())    /*the block of code is used to convert the code into json format*/
            .then(data =>{
                if(data.redirect){//&& data.Date && data.Time && data.Location && data.Destination && data.Items
                    window.location.href = data.redirect
                }
            })
            .catch(err => console.log("Error", err));
        })
